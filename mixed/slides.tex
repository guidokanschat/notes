\begin{frame}
  \frametitle{The $P_2$ element in barycentric coordinates}
  \begin{columns}
    \begin{column}{.5\textwidth}

      \begin{xalignat*}2
        \phi_{ii} &= 2\lambda_i^2 - \lambda_i,

      \end{xalignat*}
    \end{column}
  \end{columns}
