\begin{Lemma}{lmm-one-stability}
  Assume that an LMM is stable. Then, there exists a vector norm on
  $\C^{\lmms d}$ such that the operator norm of the matrix $A$
  satisfies
  \begin{gather}
    \label{eq:lmm-one-stability:1}
    \norm{A\otimes \identity} \le 1.
  \end{gather}
\end{Lemma}
